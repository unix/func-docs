### Command

All annotations used to create commands directly.

<Spacer y={1} />

export const CommandAPI = [
  {
    signature: '@Command(params: CommandParams)',
    structure: 'CommandParams = { name: string, description?: string, alias?: string }',
    description: 'create command',
  },
  {
    signature: '@Option(params: OptionParams)',
    structure:
      'OptionParams = { name: string, type?: OptionType, description?: string, alias?: string }',
    description: 'create option',
  },
  {
    signature: '@SubOptions(params: Array<OptionParams>)',
    structure: 'ditto',
    description: (
      <span>
        create a suboption, usually used after <Code>Command</Code>
      </span>
    ),
  },
  {
    signature: '@CommandMissing()',
    structure: '-',
    description: 'create a method to capture undeclared commands',
  },
  {
    signature: '@CommandMajor()',
    structure: '-',
    description: 'main command',
  },
]

<Table data={CommandAPI}>
  <Table.Column prop="signature" label="Signature" />
  <Table.Column prop="structure" label="Structure" />
  <Table.Column prop="description" label="Description" />
</Table>
